import pandas as pd

panel = pd.read_csv('data/analysis_panel.csv')

# For a regression to identify effects, you need VARIATION in the treatment
# Let's see how much variation exists in passive vs active kappa

print('--- Passive kappa (Vanguard + BlackRock + StateStreet + Passive) ---')
print(f'Mean:    {panel.contrib_PassiveTotal_lag.mean():.4f}')
print(f'SD:      {panel.contrib_PassiveTotal_lag.std():.4f}')
print(f'CV:      {panel.contrib_PassiveTotal_lag.std()/panel.contrib_PassiveTotal_lag.mean():.3f}  (SD/mean — lower = less variation)')
print(f'Min:     {panel.contrib_PassiveTotal_lag.min():.4f}')
print(f'Max:     {panel.contrib_PassiveTotal_lag.max():.4f}')
print(f'p5-p95:  {panel.contrib_PassiveTotal_lag.quantile(0.05):.4f} to {panel.contrib_PassiveTotal_lag.quantile(0.95):.4f}')

print()
print('--- Active kappa ---')
print(f'Mean:    {panel.contrib_Active_lag.mean():.4f}')
print(f'SD:      {panel.contrib_Active_lag.std():.4f}')
print(f'CV:      {panel.contrib_Active_lag.std()/panel.contrib_Active_lag.mean():.3f}')
print(f'Min:     {panel.contrib_Active_lag.min():.4f}')
print(f'Max:     {panel.contrib_Active_lag.max():.4f}')

print()
print('=== WHY THIS MATTERS: THE WITHIN-PAIR VARIATION AFTER FE ===')
print()
# After pair FE, only the deviation from each pair's OWN mean matters
# If passive kappa barely changes for a given pair, we're regressing noise on noise

pair_stats = panel.groupby('pair_id').agg(
    passive_mean=('contrib_PassiveTotal_lag', 'mean'),
    passive_sd=('contrib_PassiveTotal_lag', 'std'),
    active_mean=('contrib_Active_lag', 'mean'),
    active_sd=('contrib_Active_lag', 'std'),
    n=('return_corr', 'count')
).reset_index()

# Within-pair variation (after pair FE, this is what the regression uses)
print(f'Average within-pair SD of passive kappa: {pair_stats.passive_sd.mean():.5f}')
print(f'Average within-pair SD of active  kappa: {pair_stats.active_sd.mean():.5f}')
print()
print(f'As % of mean:')
print(f'  Passive: {pair_stats.passive_sd.mean()/pair_stats.passive_mean.mean()*100:.1f}% of mean passive kappa')
print(f'  Active:  {pair_stats.active_sd.mean()/pair_stats.active_mean.mean()*100:.1f}% of mean active kappa')

print('All 72 firms are large S&P 500 constituents.')

print()
