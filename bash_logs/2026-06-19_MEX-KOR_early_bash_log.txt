import json, glob
from collections import Counter

reasons = Counter()
total_nodes = 0
total_convos = 0

for fpath in sorted(glob.glob('/Users/aki/chatgpt_export/conversations*.json')):
    with open(fpath) as f:
        conversations = json.load(f)
    total_convos += len(conversations)
    for convo in conversations:
        mapping = convo.get('mapping', {})
        for node_id, node in mapping.items():
            total_nodes += 1
            msg = node.get('message')
            if not msg:
                reasons['no message (structural node)'] += 1
                continue
            author = msg.get('author', {})
            role = author.get('role')
            if role not in ('user', 'assistant'):
                reasons[f'role={role}'] += 1
                continue
            if msg.get('create_time') is None:
                reasons['no create_time'] += 1
                continue
            content = msg.get('content', {})
            parts = content.get('parts', [])
            text = '\n'.join(p for p in parts if isinstance(p, str))
            if not text:
                ctype = content.get('content_type', 'unknown')
                reasons[f'empty text (content_type={ctype})'] += 1
                continue
            reasons['KEPT'] += 1

print(f'Total conversations: {total_convos}')
print(f'Total mapping nodes: {total_nodes}')
print()
for reason, count in reasons.most_common():
    print(f'{count:7d}  {reason}')
"
